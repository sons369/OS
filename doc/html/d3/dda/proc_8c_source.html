326 // Per-CPU process scheduler.
327 // Each CPU calls scheduler() after setting itself up.
328 // Scheduler never returns. It loops, doing:
329 // - choose a process to run
330 // - swtch to start running that process
331 // - eventually that process transfers control
332 // via swtch back to the scheduler.
333 void scheduler(void)
334 {
335  struct proc *p;
336  struct cpu *c = mycpu();
337  c->proc = 0;
338 
339  for (;;)
340  {
341  // Enable interrupts on this processor.
342  sti();
343 
344  // Loop over process table looking for process to run.
345  acquire(&ptable.lock);
346  for (p = ptable.proc; p < &ptable.proc[NPROC]; p++)
347  {
348  if (p->state != RUNNABLE)
349  continue;
350 
351  // Switch to chosen process. It is the process's job
352  // to release ptable.lock and then reacquire it
353  // before jumping back to us.
354  c->proc = p;
355  switchuvm(p);
356  p->state = RUNNING;
357 
358  swtch(&(c->scheduler), p->context);
359  switchkvm();
360 
361  // Process is done running for now.
362  // It should have changed its p->state before coming back.
363  c->proc = 0;
364  }
365  release(&ptable.lock);
366  }
367 }

427 {
